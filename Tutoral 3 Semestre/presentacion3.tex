\section{Introducción}
\begin{frame}
    \frametitle{Introducción}

    \begin{block}{Sistemas Cuánticos Abiertos y Fluorescencia Atómica}
        Un átomo acoplado a un entorno con un gran número de modos del campo electromagnético experimenta \textbf{emisión espontánea}.\\
        \vspace{0.5em}
        Este fenómeno es una manifestación de la \textbf{disipación}: el sistema (átomo) transfiere irreversiblemente energía al entorno (modos del vacío cuántico).
    \end{block}

    \begin{columns}[T]
        \column{0.48\textwidth}
        \begin{block}{Descripción del Estado}
            El estado ya no es una función de onda pura \( |\psi\rangle \), sino un \textbf{operador densidad} \(\rho\) que describe un estado mixto:
            \[
            \rho = \overline{|\psi\rangle\langle\psi|}
            \]
            Permite describir \textbf{decoherencia} y pérdida de información cuántica hacia el entorno.
        \end{block}

        \column{0.48\textwidth}
        \begin{block}{Evolución Temporal}
            Se modela mediante:
            \begin{itemize}
                \item \textbf{Ecuación maestra de Lindblad}
                \item \textbf{Trayectorias cuánticas} (enfoque estocástico)
            \end{itemize}
            Estos formalismos capturan la \textbf{evolución no unitaria} y \textbf{disipativa}.
        \end{block}
    \end{columns}


\end{frame}


\begin{frame}{Trayectorias Cuánticas}
    
        \centering
        \textbf{La información extraída determina la dinámica observada}

    \begin{columns}[T]
        \begin{column}{0.48\textwidth}
            \begin{exampleblock}{Trayectorias de Saltos Cuánticos}
                \textbf{Medición:} Conteo de fotones discretos
                
                \textbf{Información:} \textcolor{blue}{¿CUÁNDO?} se emiten fotones
                
                \textbf{Dinámica:} 
                \begin{itemize}
                    \item Evolución suave entre saltos
                    \item Colapsos abruptos al detectar fotón
                    \item Estados puros condicionados
                \end{itemize}
                
                \centering
                \includegraphics[width=0.8\textwidth]{Tutoral 3 Semestre/tjsaltosej.png}
            \end{exampleblock}
        \end{column}
        
        \begin{column}{0.48\textwidth}
            \begin{alertblock}{Trayectorias Difusivas (Heterodina/Homodina)}
                \textbf{Medición:} Fotocorriente continua
                
                \textbf{Información:} \textcolor{red}{¿CÓMO?} fluctúa la amplitud del campo
                
                \textbf{Dinámica:}
                \begin{itemize}
                    \item Evolución continua y suave
                    \item Ruido de medición tipo Wiener
                    \item Empujones infinitesimales
                \end{itemize}
                
                \centering
                \includegraphics[width=0.8\textwidth]{Tutoral 3 Semestre/tjdifusivaej.png}
            \end{alertblock}
        \end{column}
    \end{columns}
    
    \vspace{0.5cm}
    
    \begin{block}{Mismo Sistema, Diferentes Observables}
        \begin{center}
            \textbf{Sistema Cuántico} $\rightarrow$ \textbf{Medición} $\rightarrow$ \textbf{Trayectoria Condicionada}
            
            \vspace{0.2cm}
            \small\textit{La fluorescencia del átomo nos revela caminos diferentes dependiendo de qué preguntamos}
        \end{center}
    \end{block}
\end{frame}

\begin{frame}{El Poder de la Información en Medición Cuántica}
    \begin{columns}[T]
        \begin{column}{0.6\textwidth}
            \begin{block}{De la Ecuación Maestra a Trayectorias}
                \[
                \dot{\rho} = \mathcal{L}\rho \quad \Rightarrow \quad d|\psi_c\rangle = \text{(Evolución condicionada)}
                \]
                
                \textbf{Descomposición de la ignorancia:}
                \begin{itemize}
                    \item Ecuación maestra: conocimiento promedio
                    \item Trayectorias: conocimiento individual
                    \item Información extraída $\equiv$ ruido eliminado
                \end{itemize}
            \end{block}
            
            \begin{exampleblock}{Fotodetección como Fuente de Información}
                \begin{itemize}
                    \item Cada fotón detectado $\approx$ 1 bit de información
                    \item Fotocorriente continua $\approx$ flujo de información
                    \item \textbf{Resultado:} Estado mejor conocido $\Rightarrow$ dinámica menos aleatoria
                \end{itemize}
            \end{exampleblock}
        \end{column}
        
        \begin{column}{0.38\textwidth}
            \centering
            \includegraphics[width=\textwidth]{esquema_medicion.png}
            \begin{tikzpicture}
                \node[draw, fill=blue!20, rounded corners, text width=3cm] {
                    \small\textbf{Regla de Oro:}\\
                    \textit{Más información $\Rightarrow$ Menos decoherencia}\\
                    \textit{Menos información $\Rightarrow$ Más decoherencia}
                };
            \end{tikzpicture}
        \end{column}
    \end{columns}
    
    \vspace{0.5cm}
    
    \begin{alertblock}{Aplicación en Fluorescencia Intermitente}
        \begin{center}
            El fenómeno de shelving se manifiesta \textbf{diferentemente} según el esquema de medición:
            
            \textbf{Saltos}: Periodos oscuros $\equiv$ ausencia total de fotones
            
            \textbf{Heterodina}: Periodos oscuros $\equiv$ señal ruidosa cerca de cero
        \end{center}
    \end{alertblock}
\end{frame}

\begin{frame}{Interpretación Física: Qué nos Dice la Luz Fluorescente}

                    \item Anchura: tasa de decaimiento

            \end{block}
        \end{column}
        
        \begin{column}{0.55\textwidth}
            \begin{block}{Información Temporal}
                \centering
                \begin{tabular}{c|c}
                \textbf{Medición} & \textbf{Información} \\
                \hline
                Fotoconteo & Tiempos de emisión \\
                Homodina & Amplitud del campo \\
                Heterodina & Fase y amplitud \\
                \hline
                \end{tabular}
                
                \vspace{0.5cm}
                
                \textbf{Correlaciones Temporales:}
                \begin{itemize}
                    \item $g^{(2)}(\tau)$: estadística de fotones
                    \item $S(\omega)$: espectro de potencia
                    \item Distribuciones de tiempos de espera
                \end{itemize}
            \end{block}
            
            \begin{alertblock}{Fenómeno Clave: Antibunching Cuántico}
                \[
                g^{(2)}(0) < 1 \quad \Rightarrow \quad \text{Los fotones se repelen}
                \]
                
                \small Evidencia directa de la naturaleza cuántica del emisor
            \end{alertblock}
        \end{column}
    \end{columns}
\end{frame}

\begin{frame}{Sistemas Cuánticos Abiertos}
    \begin{block}{Ecuación Maestra vs. Trayectorias Estocásticas}
        \begin{itemize}
            \item \textbf{Ecuación maestra}: Describe el comportamiento promedio del ensamble
            \item \textbf{Trayectorias estocásticas}: Evolución de sistemas individuales condicionados a mediciones
            \item \textit{Unravelings}: Descomposiciones estocásticas de la ecuación maestra
        \end{itemize}
    \end{block}
    
    \begin{exampleblock}{Relevancia Física}
        Solo algunos \textit{unravelings} corresponden a esquemas de medición físicamente realizables:
        \begin{itemize}
            \item Detección de conteo de fotones (saltos cuánticos)
            \item Detección homodina/heterodina (trayectorias difusivas)
        \end{itemize}
    \end{exampleblock}
\end{frame}

\section{Objetivos}
\begin{frame}{Objetivos del Proyecto}
    \begin{block}{Objetivo General}
        Simular sistemas cuánticos de 2 y 3 niveles mediante Teoría de Trayectorias Cuánticas para estudiar fluorescencia intermitente.
    \end{block}
    
    \begin{block}{Objetivos Específicos}
        \begin{itemize}
            \item Simular sistema de 3 niveles con estructura de \textit{shelving}
            \item Análisis fotoestadístico de tiempos de retorno al estado metaestable
            \item Implementar SDEs para trayectorias difusivas y de saltos
            \item Generar pseudo-esfera de Bloch para visualización
            \item Validar con ecuación maestra
        \end{itemize}
    \end{block}
\end{frame}
