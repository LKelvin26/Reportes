\begin{abstract}
Here, we obtain analytical results for the density operator, photon correlations, 
and full spectrum of resonance fluorescence of a three-level atom with only 
one driven transition and two decay channels from the excited state to the 
ground state. We use both direct and fluctuation operator approaches. These 
notes were the starting point for the paper H.~M. Castro-Beltran, R. 
Roman-Ancheyta, and L. Gutierrez, Phys. Rev. A, \textbf{93}, 033801 
(2016). Some calculations are found in only one of these documents.

\end{abstract}

\section{Atom-Laser Interaction}
%
\begin{figure}[b]
\includegraphics{3LA_onelaser.pdf} 
\caption{\label{fig:3LA} 
Scheme of the three-level atom-laser 
interaction and spontaneous decay rates. }
\end{figure}
%% 
We consider a single three-level atom where only one transition is driven 
by a laser, but the excited state $|e \rangle$ has two decay channels: one 
directly to the ground state $|g \rangle$, and one via a long-lived 
intermediate (ancilla) state $|a \rangle$, see Fig.~\ref{fig:3LA}. In the latter, 
the excited state decays at the rate $\gamma_d$ to $|a \rangle$ and from 
this to the ground state at the rate $\gamma_a$. 

With changes in notation, we reproduce results from Ref.~\cite{EvKe02} 
and generate many new ones reported in Ref.\cite{CaRG16}. 

The Hamiltonian and master equation are, in the frame rotating at the laser 
frequency, 
%
\begin{eqnarray}
\mathcal{H} &=& \Delta \, \sigma_{eg} \sigma_{ge} 
	+\frac{\Omega}{2} ( \sigma_{eg} +\sigma_{ge}) 	 \,, 		\\
%
\dot{\rho}(t) &=& -i \Delta [\sigma_{eg} \sigma_{ge} ,\rho] 
	-i \frac{\Omega}{2} [\sigma_{eg} +\sigma_{ge} ,\rho]    \nonumber \\ 
&& +\frac{\gamma}{2} (2\sigma_{ge} \rho \sigma_{eg} 
  -\sigma_{eg} \sigma_{ge} \rho  -\rho \sigma_{eg} \sigma_{ge})     
  \nonumber \\
&& + \frac{\gamma_d}{2} (2\sigma_{ae} \rho \sigma_{ea} 
  -\sigma_{ea} \sigma_{ae} \rho -\rho \sigma_{ea} \sigma_{ae}) 	\nonumber \\ 
&& +\frac{\gamma}{2} (2\sigma_{ga} \rho \sigma_{ag} 
  -\sigma_{ag} \sigma_{ga} \rho  -\rho \sigma_{ag} \sigma_{ga}) \,, 
\end{eqnarray}
%%
where $\Omega$ is the Rabi frequency, $\Delta$ is the detuning of the laser 
from the atomic transition, and $\sigma_{\pm}^j$ are Pauli spin operators for 
the undriven transitions. In the notation used in \cite{Carm99}, the $rho$'s 
and $sigma$'s wear a tilde to indicate they are in the frame rotating at the 
laser frequency. For simplicity, \textit{we do not use such tildes}. 

We need a complete set of spin operators in projector form: 
%
\begin{eqnarray*}
\sigma_{jk} &=& |j \rangle\langle k| \,, 
	\qquad \sigma_{kj} = \sigma_{jk}^+ \,, \qquad j,k=g,e,a   
\end{eqnarray*}
%%
from which we derive some important products: 
%
\begin{eqnarray*}
\sigma_{jk} \sigma_{lm} &=& \sigma_{jm} \delta_{kl}  \,, 
\qquad  j,k,l,m =g,e,a 	\,, 
\end{eqnarray*}
%%
allowing to simplify the master equation as 
%
\begin{eqnarray}
\dot{\rho}(t) &=& -i \Delta [\sigma_{ee} ,\rho] 
	-i \frac{\Omega}{2} [\sigma_{eg} +\sigma_{ge} ,\rho]    \nonumber \\ 
&& -	\frac{(\gamma +\gamma_d)}{2} (\sigma_{ee} \rho +\rho \sigma_{ee}) 
    -\frac{\gamma_a}{2} (\sigma_{aa} \rho +\rho \sigma_{aa}) 
        \nonumber \\
&& +\gamma \rho_{ee} \sigma_{gg}  +\gamma_d \rho_{ee} \sigma_{aa}    
    +\gamma_a \rho_{aa} \sigma_{gg}  \,.     \label{masterEqC}
\end{eqnarray}
%%

We obtain two sets of equations. The first one is:  
%
\begin{eqnarray} 	\label{eq:dotrho_jk}
\dot{\rho}_{eg} &=& i\frac{\Omega}{2} (\rho_{ee} -\rho_{gg})
 -\left( \frac{\gamma_+}{2}  +i\Delta \right) \rho_{eg} \,, \\
%
\dot{\rho}_{ge} &=& -i\frac{\Omega}{2} (\rho_{ee} -\rho_{gg})
 -\left( \frac{\gamma_+}{2}  -i\Delta \right) \rho_{ge} 	\,,
 	\nonumber 	\\
%
\dot{\rho}_{ee} &=& -i\frac{\Omega}{2} (\rho_{ge} -\rho_{eg})
 	-\gamma_+ \rho_{ee} 	\,,	\nonumber 	\\
%
\dot{\rho}_{gg} &=& i\frac{\Omega}{2} (\rho_{ge} -\rho_{eg})  
	+\gamma_- \rho_{ee} -\gamma_a \rho_{gg} 
	+\gamma_a \,.	 	\nonumber 	
\end{eqnarray}
%%
where
%
\begin{eqnarray}
\gamma_+ &=& \gamma + \gamma_d \,, 	\qquad 
\gamma_- = \gamma - \gamma_a \,.
\end{eqnarray}
%%
This system of equations is a reduced one since the equation 
$\dot{\rho}_{aa} = \gamma_d \rho_{ee} -\gamma_a \rho_{aa}$ was 
eliminated due to the conservation of population, 
$\rho_{gg} +\rho_{ee} +\rho_{aa} =1$, replaced in the equation 
$\dot{\rho}_{gg} = i (\Omega/2) (\rho_{ge} -\rho_{eg})  
	+\gamma \rho_{ee} +\gamma_a \rho_{aa}$.

The other set of equations involves the coherences linking the states 
$|e \rangle$ and $|g \rangle$ to the state $|a \rangle$, 
%
\begin{eqnarray*}
\dot{\rho}_{ga} &=& -i\frac{\Omega}{2} \rho_{ea} 
	-\frac{\gamma_a}{2} \rho_{ga} 	\,,\\
\dot{\rho}_{ea} &=& -i\frac{\Omega}{2} \rho_{ga} -\left( i\Delta 
	+\frac{\gamma +\gamma_d +\gamma_a}{2}  \right) \rho_{ea} \,, \\
\dot{\rho}_{\nu \mu} &=& \dot{\rho}_{\mu \nu}^{\ast} 	\,.
\end{eqnarray*}
%%
These coherences have zero mean and vanish for times 
$t \sim \gamma^{-1}$. The two sets are decoupled; only the first one 
is relevant for the observables of interest, which involve only the driven 
transition. 

Noticing that $\langle \sigma_{jk} \rangle =\rho_{kj}$ we rewrite the 
Bloch equations  (\ref{eq:dotrho_jk}) as 
%
\begin{eqnarray} 	\label{eq:dotsigma_jk}
\langle \dot{\sigma}_{-} \rangle 
&=& -\left( \frac{\gamma_+}{2} +i\Delta \right) 
	\langle \sigma_{-} \rangle 
 +i\frac{\Omega}{2} (\langle \sigma_{ee} \rangle 
 	-\langle \sigma_{gg} \rangle)	\,,  \nonumber  	\\
%
\langle \dot{\sigma}_{+} \rangle 
&=& -\left( \frac{\gamma_+}{2} -i\Delta \right) 
	\langle \sigma_{+} \rangle 
 -i\frac{\Omega}{2} (\langle \sigma_{ee} \rangle -\langle 
 	\sigma_{gg} \rangle)  \,,	\nonumber 	\\
%
\langle \dot{\sigma}_{ee} \rangle 
&=& i\frac{\Omega}{2} (\langle \sigma_{-} \rangle 
	-\langle \sigma_{+} \rangle)
 	-\gamma_+ \langle \sigma_{ee} \rangle	\,, \nonumber 	\\
%
\langle \dot{\sigma}_{gg} \rangle 
&=& -i\frac{\Omega}{2} (\langle \sigma_{-} \rangle 
	-\langle \sigma_{+} \rangle)  
	+\gamma_- \langle \sigma_{ee} \rangle  	
	 -\gamma_a \langle \sigma_{gg} \rangle  +\gamma_a \,.	
	\nonumber \\	
\end{eqnarray}
%%

We write this set in compact form as 
%
\begin{subequations}
\begin{eqnarray}
\langle \dot{\mathbf{s}} \rangle 
	&=& \mathbf{M} \langle \mathbf{s} \rangle +\mathbf{b} 	\,, \\ 
\mathbf{s} &\equiv&   
	\left( \sigma_-,  \sigma_+, \sigma_{ee}, \sigma_{gg}  \right)^T 	\,, \\ 
\mathbf{b} &=& (0,0,0,\gamma_a)^T 	\,,  
\end{eqnarray}
\end{subequations}
%%
%
\begin{eqnarray} 	\label{eq:matrixM}
\mathbf{M} &=& \left( \begin{array}{cccc}
-\frac{1}{2}(\gamma_+ +i2\Delta) & 0 & i\Omega/2 & -i\Omega/2 \\ 
0 & -\frac{1}{2}(\gamma_+ -i2\Delta) & -i\Omega/2 & i\Omega/2 	 \\ 
i\Omega/2 & -i\Omega/2 & -\gamma_+ & 0 		\\ 
-i\Omega/2 & i\Omega/2 & \gamma_- & -\gamma_a 
\end{array}  \right) 	\nonumber \\ 
\end{eqnarray}
%%
Formal solutions for the Bloch equations 
%
\begin{equation}
\frac{d}{dt} \left( \langle \mathbf{s} \rangle 
	+\mathbf{M}^{-1} \mathbf{b} \right) = 
\mathbf{M} \left( \langle \mathbf{s} \rangle 
	+\mathbf{M}^{-1}\mathbf{b} \right) 	
\end{equation}
%%
are given by 
%
\begin{eqnarray}
\langle \mathbf{s}(t) \rangle &=& e^{\mathbf{M} t} \left[ 
 	\langle \mathbf{s}(0) \rangle +\mathbf{M}^{-1} \mathbf{b} \right] 
	-\mathbf{M}^{-1} \mathbf{b} 	\\
&=& S^{-1} e^{\lambda t} S \left[ 
	\langle \mathbf{s}(0) \rangle +\mathbf{M}^{-1} \mathbf{b} \right] 
	-\mathbf{M}^{-1} \mathbf{b} 	\,, \nonumber 
\end{eqnarray}
%%
where $S$ is a matrix that diagonalizes $\mathbf{M}$, whose 
eigenvalues are given in the matrix 
$\lambda= diag(\lambda_1, \lambda_2, \lambda_+, \lambda_-)$. 
An exact general analytical solution is not possible. 

It looks hopeless to perform a full matrix treatment (inverse, eigenvectors, 
eigenvalues, diagonalization matrix) analytically. Even the on-resonance 
case, $\Delta=0$, is complicated, see Section \ref{sec:MatrixEigensystem}. 
Only one eigenvalue and its corresponding eigenvector can be obtained 
easily: $\lambda_1 =-\gamma_+/2$, $\mathbf{r}_1 =(1,1,0,0)^T$. 
Mathematica gives very complicated exact expressions for the other 
eigenvalues and eigenvectors.

However, approximate solutions, with simple expressions of the eigenvalues 
for $\Delta=0$, are obtained later.

\section{Steady State of Density Operator} 
We start with the equations for $\rho$ including $\rho_{aa}$,
%
\begin{eqnarray}
\dot{\rho}_{ee} &=& 0 = -i\frac{\Omega}{2} (\rho_{ge} -\rho_{eg})
 	-\gamma_+ \rho_{ee}	\\ 
\dot{\rho}_{aa} &=& 0 = \gamma_d \rho_{ee} -\gamma_a \rho_{aa} \\
\dot{\rho}_{gg} &=& 0 = i\frac{\Omega}{2} (\rho_{ge} -\rho_{eg})  
 +\gamma_- \rho_{ee} -\gamma_a \rho_{gg} +\gamma_a   \\
\dot{\rho}_{ge} &=& 0 = -i\frac{\Omega}{2} (\rho_{ee} -\rho_{gg})
	-\left( \frac{\gamma_+}{2} -i\Delta \right) \rho_{ge} 	\\
\dot{\rho}_{eg} &=& 0 = i\frac{\Omega}{2} (\rho_{ee} -\rho_{gg})
	-\left( \frac{\gamma_+}{2} +i\Delta \right) \rho_{eg} 	\,.
\end{eqnarray}
%%

For the rest of this Section we mean $\rho$ to be in the steady 
state ($\rho^{st}$). Adding B1 and B3 
%
\begin{eqnarray}
0 &=& -\gamma_+ \rho_{ee} - \gamma_- \rho_{ee} 
	-\gamma_a \rho_{gg} +\gamma_a 		\nonumber \\
\to \rho_{gg} &=& 1- \frac{\gamma_d +\gamma_a}{\gamma_a} \rho_{ee} \,.
\end{eqnarray}
%%

Inserting B6 in B4
%
\begin{eqnarray*}
0 &=& -\left( \frac{\gamma_+}{2} -i\Delta \right) \rho_{ge} \\
    && -i\frac{\Omega}{2} \left[ \rho_{ee} 
    -\left( 1- \frac{\gamma_d +\gamma_a}{\gamma_a}  \rho_{ee} \right)
    \right]     \\
&=& -\left( \frac{\gamma_+}{2} -i\Delta \right) \rho_{ge} 
    -i\frac{\Omega}{2} \left[  (2+q) \rho_{ee} -1 \right]     \\ 
\rho_{ge} &=& 
   \frac{ -i\Omega \left[ (2+q) \rho_{ee} -1 \right]}{\gamma_+ -i2\Delta} \,,    
       \qquad \quad \rho_{eg} = \rho_{ge}^{\ast}   \,,
\end{eqnarray*}
%%
donde $q= \gamma_d /\gamma_a$. Then
%
\begin{eqnarray}
\rho_{ge} -\rho_{eg} &=& 
	- \frac{ i 2\gamma_+ \Omega \left[ (2+q) \rho_{ee} -1 \right]}
       {\gamma_+^2 +4\Delta^2}     	\nonumber \\
-i \frac{\Omega}{2} \left( \rho_{ge} -\rho_{eg} \right) 
&=& - \gamma_+ \Omega^2  \left[ \frac{ (2+q) \rho_{ee} -1 }
       {\gamma_+^2 +4\Delta^2} \right]     
\end{eqnarray}
%%
We used $q=\gamma_d/\gamma_a$. Substituting B7 into B1 
%
\begin{eqnarray}
0 &=& \frac{ \Omega^2 \gamma_+}{\gamma_+^2 +4\Delta^2} 
       -\gamma_+ \rho_{ee}     
- \frac{ \Omega^2 \gamma_+ (2+q) \rho_{ee} }
       {\gamma_+^2 +4\Delta^2}      \nonumber  \\ 
&=& -\left[ 1+  \frac{ \Omega^2 (2+q) }
       {\gamma_+^2 +4\Delta^2} \right] \rho_{ee} 
	+\frac{ \Omega^2 }{\gamma_+^2 +4\Delta^2}    \nonumber  \\
&=& -\left[ \gamma_+^2 +4\Delta^2 +\Omega^2 (2+q) 
    \right] \rho_{ee} +\Omega^2 		 \nonumber \\
\rho_{ee}^{st} &=& \frac{\Omega^2}{ (2+q) \Omega^2  
    +\gamma_+^2 +4\Delta^2 } 	\,.
\end{eqnarray}
%%

Substituting B8 into B2 
%
\begin{eqnarray} 
0 &=& \gamma_d \rho_{ee} -\gamma_a \rho_{aa} 
    \qquad \to \rho_{aa} =q \rho_{ee}   \nonumber  \\
\rho_{aa}^{st} &=& 
	\frac{q\Omega^2}{ (2+q) \Omega^2  +\gamma_+^2 +4\Delta^2 } \,.
\end{eqnarray}
%%

Substituting B8 and B9 into $1=\rho_{ee} +\rho_{aa} +\rho_{gg}$
%
\begin{eqnarray} 
\rho_{gg}^{st} &=& 1- \rho_{ee} -\rho_{aa} = 1-(1+q) \rho_{ee}     
    \nonumber     \\
&=& 1- \frac{ (1+q) \Omega^2}{ (2+q) \Omega^2  
    +\gamma_+^2 +4\Delta^2 }     \nonumber \\
&=& \frac{ \Omega^2 +\gamma_+^2 +4\Delta^2 }
    { (2+q) \Omega^2 +\gamma_+^2 +4\Delta^2 } 	\,.
\end{eqnarray}
%%

Substituting B8 and B10 into B4, 
%
\begin{eqnarray*}
\left( \frac{\gamma_+}{2} -i\Delta \right) \rho_{ge}  
&=& -i\frac{\Omega}{2} (\rho_{ee} -\rho_{gg})   \\ 
(\gamma_+  -i2\Delta) \rho_{ge} 
 &=& \frac{ i \Omega [\gamma_+^2 +4\Delta^2] }
    { (2+q) \Omega^2 +\gamma_+^2 +4\Delta^2 }    \\ 
&=& \frac{ i \Omega (\gamma_+  -i2\Delta) (\gamma_+  +i2\Delta)}
    { (2+q) \Omega^2 +\gamma_+^2 +4\Delta^2 }    \\ 
\end{eqnarray*}
%%
%
\begin{eqnarray}
\rho_{ge}^{st} &=& \frac{ -\Omega [2\Delta -i \gamma_+] }
  	 { (2+q) \Omega^2 +\gamma_+^2 +4\Delta^2 }    \\ 
\rho_{eg}^{st} &=& \frac{ -\Omega [2\Delta +i \gamma_+] }
  	{ (2+q) \Omega^2 +\gamma_+^2 +4\Delta^2 } 	\,.
\end{eqnarray}
%%

Defining 
$\alpha_- =\langle \sigma_- \rangle_{st} =\rho_{eg}^{st} $, 
$\alpha_+ =\alpha_-^{\ast}$, 
$\alpha_{ee} =\langle \sigma_+ \sigma_- \rangle_{st} =\rho_{ee}^{st}$, 
$\alpha_{gg} =\langle \sigma_- \sigma_+ \rangle_{st} =\rho_{gg}^{st} $, 
$\alpha_{aa} =\langle \sigma_+^a \sigma_-^a \rangle_{st} =\rho_{aa}^{st}$, 
the steady state of the relevant elements of the density matrix are  
%
\begin{subequations}
\begin{eqnarray}
\alpha_{\mp} &=& \frac{-\Omega[2\Delta \pm i \gamma_+]}{D} \,,  \\ 
\alpha_{gg} &=& \frac{\Omega^2 +\gamma_+^2 +4\Delta^2}{D} 
	\,, \\ 
\alpha_{ee} &=& \frac{\Omega^2}{D} \,, 	\\ 
\alpha_{aa} &=& q\frac{ \Omega^2}{D} \,, 	 
\end{eqnarray}
%%
where 
%
\begin{eqnarray}
D &=& (2 +q)\Omega^2 +\gamma_+^2 +4\Delta^2 	\,,  \\
D_0 &=& (2+q)\Omega^2 +\gamma_+^2 \,, 	\qquad 	\Delta=0 \,, 
	\nonumber
\end{eqnarray}
\end{subequations}
%%
and
%
\begin{eqnarray}
q = \gamma_d/\gamma_a 	\,.
\end{eqnarray}
%%
For $\gamma_d=0$, we recover the 2LA results. Evers and Keitel 
\cite{EvKe02} find that $\gamma_d \sim 3.3 \gamma_a$ is very close 
to the optimum value for the appearance of the extra narrow peak.

\section{Shelving: Bright and Dark Periods} 
This system features long bright and dark periods if 
$\gamma \gg \gamma_d\,, \gamma_a$. We outline the random telegraph 
model (RTM) approach to calculate the average bright and dark periods,  
proposed by Pegg and Knight \cite{PeKn88a,PeKn88b,EvKe02}. An 
alternative approach is given by Cohen-Tannoudji and Dalibard  
\cite{CTDal86}.  

Let us assume the atom is initially in the $| g \rangle - | e \rangle$ subspace 
so that $\rho_{aa}(0)=0$. We will find the expectation value $T_B$ for the 
atom to stay in the $| g \rangle - | e \rangle$ subspace without moving into 
$| a \rangle$. If the atom is still in this subspace at a later time $t=t_1$, the 
life expectancy at this time is again $T_B$. If $P_{ge}(t_1)$ is the probability 
of remaining in the $| g \rangle - | e \rangle$ subspace without moving into 
$| a \rangle$ in the time $t_1$, we have  
%
\begin{eqnarray}
T_B &=& P_{ge}(t_1) (t_1+ T_B) +[1- P_{ge}(t_1)] f t_1 \,, 
\end{eqnarray}
%%
where $0 \leq f \leq 1$. The second term is the probability of not staying 
in the above subspace multiplied by the time $f t_1$ at which the atom 
moves to the state $| a \rangle$. Thus, 
%
\begin{eqnarray}
T_B &=& \frac{t_1 P_{ge}(t_1)}{ 1- P_{ge}(t_1) } +f t_1  
\end{eqnarray}
%%
(we write this step differently from \cite{PeKn88a,PeKn88b,EvKe02}, etc). 
For large $t_1$, the atom may jump out of $| g \rangle - | e \rangle$ and 
back into it. To avoid multiple transitions between subspaces we make 
$t_1 \to 0$ so that $P_{ge} (t_1) \to 1$ and 
%
\begin{eqnarray}
\lim_{t_1 \to 0} [1- P_{ge} (t_1)] &=& \lim_{t_1 \to 0} \rho_{aa} (t_1) \,.
\end{eqnarray}
%%
We obtain 
%
\begin{eqnarray}
T_B &=& \lim_{t_1 \to 0} \frac{t_1}{\rho_{aa} (t_1)}  	\\ 
T_B^{-1} &=& \left( \frac{ d \rho_{aa} }{dt} \right)_{t=0} \,,
\end{eqnarray}
%%
with the condition $\rho_{aa}(0) =0$. 

A similar argument for the atom to stay continuously in $| a \rangle$ yields 
the average dark period $T_D$. Let us assume the atom is initially in the 
trapping state $| a \rangle$, $\rho_{aa}(0)=1$. If the atom is still in 
$| a \rangle$ at a later time $t=t_1$ the life expectancy at this time is again 
$T_D$. If $P_{a}(t_1)$ is the probability of remaining in $| a \rangle$ 
without moving into the $| g \rangle - | e \rangle$ subspace in the time 
$t_1$, we have  
%
\begin{eqnarray}
T_D &=& P_{a}(t_1) (t_1+ T_D) +[1- P_{a}(t_1)] f t_1 \,, 
\end{eqnarray}
%%
where $0 \leq f \leq 1$. The second term is the probability of not staying 
in $| a \rangle$ by the time $f t_1$ at which the atom moves to the 
subspace $| g \rangle - | e \rangle$. Thus, 
%
\begin{eqnarray}
T_D &=& \frac{t_1 P_{a}(t_1)}{ 1- P_{a}(t_1) } +f t_1  	\,.
\end{eqnarray}
%%
For large $t_1$, the atom may jump to $| g \rangle - | e \rangle$ and 
back into $| a \rangle$. To avoid multiple transitions between subspaces 
we make $t_1 \to 0$ so that $P_{a} (t_1) \to 1$ and 
%
\begin{eqnarray}
T_D^{-1} &=& \lim_{t_1 \to 0} \frac{ 1- P_a (t_1) }{t_1} 		\quad \to 
 	\lim_{t_1 \to 0} \frac{ \rho_{aa} (0) - \rho_{aa} (t_1) }{t_1} 	\nonumber \\
&=& - \left( \frac{ d \rho_{aa} }{dt} \right)_{t=0} \,,
\end{eqnarray}
%%
with the condition $\rho_{aa}(0) =1$. 

Now, we apply the above results. We only need the equation for the trapping 
state, $\dot{\rho}_{aa} = \gamma_d \rho_{ee} -\gamma_a \rho_{aa}$, 
evaluated in the quasi-steady state (steady state of the two-level system of 
the bright transition), here denoted as $t=0$. For the bright period 
$\rho_{aa}(0) =0$ and $\rho_{ee}(0) \to (\rho_{ee}^{st})^{2LA}$, 
%
\begin{eqnarray}    
T_B^{-1} &=& (\dot{\rho}_{aa})_{t=0} 	\to (\rho_{ee}^{st})^{2LA} \,, \nonumber  \\  
T_B &=& \frac{2\Omega^2 +\gamma^2 +4\Delta^2}{\gamma_d \Omega^2} \,. 
\end{eqnarray}
%% 
For the dark period $\rho_{aa}(0) =1$ and $\rho_{ee}(0) \to 0$,
%
\begin{eqnarray}    
T_D^{-1} &=& (\dot{\rho}_{aa})_{t=0} \,, 	\qquad \to 
	T_D = \gamma_a^{-1} \,.
\end{eqnarray}
%% 

\textcolor{red}{How do we get them?} 
The distributions of lengths of dark and bright periods are 
%
\begin{eqnarray}    
P_D = \frac{1}{T_D} e^{-t/T_D}	 \,, 	\qquad 
	P_B = \frac{1}{T_B} e^{-t/T_B}	\,, 
\end{eqnarray}
%% 
respectively.

\subsection{Scattering Rate} 
The photon scattering rate, the frequency of photon emissions, is 
$r_s =\gamma \langle \sigma_{ee} \rangle$, which in the saturation regime 
in a bright period is $r_s^{st}=\gamma/2$. Then, the number of photons 
emitted in the  interval $[0,t]$ is 
%
\begin{eqnarray} 
N(t) = \gamma \int_0^t \langle \sigma_{ee} (t') \rangle \, dt' \,.
\end{eqnarray}
%%
Equivalently, we may calculate the average time interval between two 
consecutive photons 
%
\begin{eqnarray} 	\label{eq:meantau}
 \bar{\tau} &=& \frac{1}{\gamma \alpha_{ee} } 
	=\frac{(2 +q)\Omega^2 +\gamma_+^2 +4 \Delta^2}
	{\gamma \Omega^2} 	\,,		
\end{eqnarray}
%% 
which is larger than the case of the 2LA, increased by the additional coupling 
of the excited state to the long-lived state. Of course, as an average time, it 
includes the effect of dark periods, which enhance the emission fluctuations. 
So, it is better to rely on photon correlations or the waiting-time distribution. 
The Mandel's Q parameter for the shelving case was studied in 
\cite{MeSc90}.

\section{Two-Time Correlations} 
In our work, we investigate several two-time correlation functions. 
The quantum regression formula allows to write equations for the 
correlation functions: 
%
\begin{subequations}
\begin{eqnarray}
\frac{d}{d \tau} \langle \sigma_{+}(0) \mathbf{s} (\tau) \rangle_{st} 
 &=& \mathbf{M} \langle \sigma_{+}(0) \mathbf{s} (\tau) \rangle_{st} 
	+\mathbf{b} \langle \sigma_{+} \rangle_{st} 	\,, \qquad 	\\ 
%
\frac{d}{d \tau} \langle \mathbf{s} (\tau) \sigma_{-}(0)  \rangle_{st} 
 &=& \mathbf{M} \langle \mathbf{s} (\tau) \sigma_{-}(0)  \rangle_{st} 
	+\mathbf{b} \langle \sigma_{-} \rangle_{st} 	\,, 
\end{eqnarray}
%%
\begin{eqnarray}
%
\frac{d}{d \tau} \langle \sigma_{+}(0) \mathbf{s} (\tau) 
	\sigma_{-}(0) \rangle_{st}  
 &=& \mathbf{M} \langle \sigma_{+}(0) \mathbf{s} (\tau) 
 	\sigma_{-}(0) \rangle_{st}  		\nonumber \\ 
	&& +\mathbf{b} \langle \sigma_{+} \sigma_{-} \rangle_{st} 	\,,
\end{eqnarray}
\end{subequations}
%%
with solutions (compacted to just one formula)
%
\begin{eqnarray}
\langle A_+ \mathbf{s}(t) A_- \rangle 
	&=& e^{\mathbf{M} t} \left[ \langle A_+ \mathbf{s}(0) A_- \rangle 
	+\mathbf{M}^{-1} \mathbf{b} \langle A_+ A_- \rangle_{st} \right] 
	\nonumber \\
	&& 	-\mathbf{M}^{-1} \mathbf{b}  \langle A_+ A_- \rangle_{st} \,.
\end{eqnarray}
%%

The above corelations have initial conditions: 
%
\begin{subequations}
\begin{eqnarray} 
\langle \sigma_+ \mathbf{s} \rangle 
  	&=& \left\{ \alpha_{ee}, 0, 0,  \alpha_+ \right\}   \,,     \\
\langle \mathbf{s} \sigma_-  \rangle 
	&=&  \left\{ 0, \alpha_{ee}, 0, \alpha_- \right\}    \,,     \\ 
\langle \sigma_+ \mathbf{s} \sigma_- \rangle 
	&=& \left\{ 0, 0, 0, \alpha_{ee}  \right\}        \,.
\end{eqnarray}
\end{subequations}

\subsection{Fluctuation Operator Approach}
We also solve for two-time correlations of fluctuation operators. From the 
Bloch equations $\langle \mathbf{\dot{s}} \rangle = 
\mathbf{M} \langle \mathbf{s} \rangle +\mathbf{b}$, the steady state 
($t \to \infty$) is 
%
\begin{equation}
0 =\mathbf{M} \langle \mathbf{s} \rangle_{st} +\mathbf{b}  \,, 
    \qquad \to \langle \mathbf{s} \rangle_{st} 
       =\mathbf{M}^{-1} \mathbf{b} \,.  \label{eq:SBlochss}	
\end{equation}
%%
We split the field (dipole) into a mean plus fluctuations, 
$\mathbf{s}(t) = \langle \mathbf{s} \rangle_{st} +\Delta \mathbf{s}(t)$ 
leading to 
%
\begin{eqnarray}
\frac{d}{dt} \langle \Delta \mathbf{s} \rangle +0
&=& \mathbf{M} \left[ \langle \mathbf{s} \rangle_{st} 
	+\langle \Delta \mathbf{s} \rangle \right] +\mathbf{b}  \nonumber \\
&=& \mathbf{M} \langle \Delta \mathbf{s} \rangle 
	+\mathbf{M} \langle \mathbf{s} \rangle_{st} +\mathbf{b}  
	\nonumber 	\\
\frac{d}{dt} \langle \Delta \mathbf{s} \rangle 
	&=& \mathbf{M} \langle \Delta \mathbf{s} \rangle  \,.
\end{eqnarray}
%%
Of course, this is meaningless because, by definition, 
$\langle \Delta \mathbf{s} \rangle =0$. We have to resort to the QRF 
to calculate two-time correlations: (i) remove the brackets, multiply 
$\Delta \mathbf{s}$ from the left and from the right by suitable operators 
$\Delta A_+$ and $\Delta A_-$, respectively, and (iii) put again the 
brackets, obtaining 
%
\begin{eqnarray}
\frac{d}{d\tau} \langle \Delta A_+ \Delta \mathbf{s}(\tau) \Delta A_- \rangle 
 &=& \mathbf{M} \langle \Delta A_+ \Delta \mathbf{s}(\tau) \Delta A_- \rangle  \,, 
 	\qquad	
\end{eqnarray}
%%
with solutions 
%
\begin{eqnarray} 	\label{eq:FlucCorrSol}
\langle \Delta A_+ \Delta \mathbf{s}(\tau) \Delta A_- \rangle 
 &=& e^{\mathbf{M} \tau} \langle \Delta A_+ \Delta \mathbf{s} 
 	\Delta A_- \rangle  	\\ 
&=& S^{-1} e^{\lambda t} S  \langle \Delta A_+ \Delta \mathbf{s} 
 	\Delta A_- \rangle 	\,.	\nonumber
\end{eqnarray}
%%
Again, there are no exact analytical solutions for the full time-dependent 
problem.

The initial conditions of the correlations are:

\lefteqn{  \langle \Delta \sigma_+ \Delta \mathbf{s} \rangle = 
\left( \begin{array}{c}  \alpha_{ee} -\alpha_+ \alpha_- \\  -\alpha_+^2 \\

\end{eqnarray}

 The first element, for example, is calculated as 
%

	[\sigma_-  - \langle \sigma_-  \rangle_{st}] \rangle_{st} 	\\

	\end{array} \right) 		\,.  \nonumber 
\end{eqnarray}

\section{Stationary Power Spectrum} 
The stationary power spectrum is given by the Wiener-Khintchine spectrum, 
which consists of the Fourier transform of the field autocorrelation function,  
%
\begin{eqnarray}
S(\omega) &=& \frac{1}{ \pi \alpha_{ee} } \mathrm{Re}
\int_0^{\infty} d\tau e^{-i \omega \tau} 
	\langle \sigma_+ (0) \sigma_- (\tau) \rangle_{st}  	\,.
\end{eqnarray}
%%
The factor $(\pi \alpha_{ee} )^{-1}$ allows to normalize the spectrum as 
$\int_{-\infty}^{\infty} S(\omega) d \omega =1$. 

To reveal more clearly the role of fluctuations, we split the atomic 
operators into a mean,  $\alpha_{jk}$, plus fluctuations, 
$\Delta \sigma_{jk}(t)$, that is,  
%
\begin{eqnarray} 	\label{eq:dipoleSplit}
\sigma_{jk} (t) = \langle \sigma_{jk}  \rangle_{st} 
	+\Delta \sigma_{jk} (t) 	\,, 
\end{eqnarray}
%%
where, by definition, $\langle \Delta \sigma_{\pm}(t)\rangle =0$,  
such that 
%
\begin{subequations}
\begin{eqnarray}
\langle \sigma_+(0) \sigma_-(\tau) \rangle_{st}  
&=& |\alpha_+ |^2 
	+\langle \Delta \sigma_+(0) \Delta \sigma_-(\tau) \rangle_{st} \,. \qquad
\end{eqnarray}
%%
At zero interval, $\tau=0$, 
%
\begin{eqnarray} 
\langle \sigma_+ \sigma_- \rangle_{st} 
	&=& |\alpha_+ |^2   
	+\langle \Delta \sigma_+ \Delta \sigma_- \rangle_{st}  \,, 
\end{eqnarray}
\end{subequations}
%%
where the first term on the rhs is the coherent part of the dipole emission, 
and the second term is the incoherent intensity, 
%
\begin{eqnarray} 
\langle \Delta \sigma_+ \Delta \sigma_- \rangle_{st} 
&=& \alpha_{ee} -|\alpha_+|^2 		 \\  
&=& \frac{ (2+q)\Omega^4}
    { [(2+q)\Omega^2 +\gamma_+^2 +4\Delta^2 ]^2} \,.   \nonumber 
\end{eqnarray}
%%

An important relation is the ratio of coherent (elastic) to incoherent 
(inelastic) intensity: 
%
\begin{eqnarray} 
\frac{I_{coh}}{I_{inc}} &=& \frac{|\langle \sigma_+  \rangle_{st}|^2}
	{\langle \Delta \sigma_+ \Delta \sigma_- \rangle_{st}}		\\ 
&=& \frac{ \Omega^2 [\gamma_+^2 +4\Delta^2]/D^2 }
	{(2+q) \Omega^4 /D^2} 
	= \frac{ \gamma_+^2 +4\Delta^2 }{(2+q) \Omega^2}  \,. 
	\nonumber 
\end{eqnarray}
%%
This is the inverse of the saturation parameter $s$, which gives the 
intensity of the driving field necessary to saturate the atom, where 
the incoherent intensity is larger than the coherent one. Later, we will 
explain some results in terms of this parameter.

The splitting (\ref{eq:dipoleSplit}) separates the spectrum in two parts: 
%
\begin{eqnarray} 	\label{eq:3LASpectrum?}
S(\omega) &=& S_{coh}(\omega) +S_{inc}(\omega) 	\,,
\end{eqnarray}
%%
where 
%
\begin{eqnarray}
S_{coh}(\omega) &=& \frac{|\alpha_+ |^2}{ \pi \alpha_{ee}  }
	\mathrm{Re} \int_0^{\infty}  e^{-i \omega \tau} d\tau 
	= \frac{|\alpha_+ |^2}{ \pi \alpha_{ee}  } 
	\delta(\omega)	\,, 	\qquad 
\end{eqnarray}
%%
is the coherent spectrum due to elastic scattering, and 
%
\begin{eqnarray} 	\label{eq:Sinc}
S_{inc}(\omega) 
&=& \frac{1}{ \pi \alpha_{ee}  } \mathrm{Re} \int_0^{\infty} 
	d\tau e^{-i \omega \tau} 
	\langle \Delta \sigma_+(0) \Delta \sigma_-(\tau) \rangle_{st} 
	\,, 	\nonumber \\
\end{eqnarray}
%%
is the incoherent (inelastic) spectrum due to atomic fluctuations.

\subsection{Coherent (delta) Spectrum and The Incoherent Narrow Peak}
The coherent spectrum is 
%
\begin{eqnarray} 	\label{eq:SCoh3LA}
S_{coh}(\omega) 
&=& \frac{ \Omega^2[\gamma_+^2 +4\Delta^2]/D^2 }
    {\pi \Omega^2/D} \delta(\omega)	    \\
   &=& \frac{ \gamma_+^2 +4\Delta^2 }
    {\pi [(2+q)\Omega^2 +\gamma_+^2 +4\Delta^2 ] } \delta(\omega)	   \,. 
    \nonumber 	 
\end{eqnarray}
%%

The total intensity emitted by the excited state should not differ from that 
of the 2LA spectrum, regardless of now having two decay channels. But 
(\ref{eq:SCoh3LA}) is smaller than the elastic peak of the 2LA spectrum 
(where $\gamma_d=q=0$). Evers and Keitel \cite{EvKe02} showed that 
the elastic peak is split into a smaller elastic peak plus a very narrow 
inelastic one, which carries the difference in intensity. This peak is due 
to the shelving of the electronic population in the long-lived state. 
Hegerfeldt and Plenio \cite{HePl95} and Garraway et al \cite{GaKK95} 
first predicted this effect for the $V$ and $\Lambda$ 3LA configurations. 
This means that we must replace Eq.(\ref{eq:3LASpectrum?}) with 
%
\begin{eqnarray} 	\label{eq:Full3LASpectrum}
S(\omega) &=& S_{coh}(\omega) +S_{ep}(\omega) 
	+S_{inc}^{Mollow}(\omega) \,.
\end{eqnarray}
%%

Writing $\pi S_{coh}^{NLA} =I^{NLA} \delta(\omega)$, $N=2,3,$ the 
relative intensity of the narrow inelastic peak is   
%
\begin{eqnarray}
I_{ep} &=& \frac{|\alpha_+^{2LA} |^2}{ \alpha_{ee}^{2LA} } 
	-\frac{|\alpha_+^{3LA} |^2}{ \alpha_{ee}^{3LA} } 	\\
&=& \frac{ \Omega^2 [q(\gamma^2 +4\Delta^2) -2\gamma_d \gamma_+]}
	{[2\Omega^2 +\gamma^2 +4\Delta^2 ]
	[(2+q)\Omega^2 +\gamma_+^2 +4\Delta^2 ] } \,. 
	\nonumber 	
%\\ &=& \frac{ \Omega^2 [\gamma^2 +4\Delta^2 
%	-2\gamma_a(\gamma  +\gamma_d)]}
%	{[2\Omega^2 +\gamma^2 +4\Delta^2 ]
%	[ \Omega^2 +\frac{1}{q}(2\Omega^2 +(\gamma +\gamma_d)^2 
%	+4\Delta^2) ] } \,. 	\nonumber 
\end{eqnarray}
%% 

The Rabi frequency for the maximum intensity of the extra peak, from 
$d I_{ep}/d \Omega =0$, is given by 
%
\begin{eqnarray}
\Omega_{opt} &=& \left[ \frac{ (\gamma^2 +4\Delta^2) 
\left[ \gamma_+^2 +4\Delta^2 \right] }{ 2(2+q) } \right]^{1/4} \,.
\end{eqnarray}
%%
\textcolor{blue}{En resonancia, si 
$(\gamma, \gamma_d, \gamma_a) =(0.95,\,0.050,\, 0.015$, $q=10/3$, 
valores cercanos a los usados en nuestros artículos (hab\'iamos 
usado $\gamma=1$), entonces $\Omega_{opt} = 0.539 \gamma$.

I_{ep} &=& \left( |\langle \sigma_{eg} \rangle_{st} |^2 \right)_{2LA}

\subsection{Incoherent (Mollow) Spectrum} 
An approximated analytical solution of the correlation 
$\langle \Delta \sigma_+(0) \Delta \sigma_-(\tau) \rangle_{st}$ in the 
limiting case of $\Delta =0$ is possible using Laplace transform, written 
as a sum of exponentials [see Eq.(\ref{eq:p_tau})],
%
\begin{eqnarray}
\langle \Delta \sigma_+(0) \Delta \sigma_-(\tau) \rangle_{st} 
	= \sum_m C_m e^{\lambda_m \tau} \,, 
\end{eqnarray}
%%
where $\lambda_m$ are the eigenvalues of $\mathbf{M}$, with negative 
real part. Hence, we have simple integrals of the type
%
\begin{eqnarray} 	%\label{eq:integrals}
\int_0^{\infty} d\tau e^{(-i \omega +\lambda_m) \tau} 
	&=& \left. \frac{e^{(-i \omega +\lambda_m) \tau}}
	{-i \omega +\lambda_m} \right|_0^{\infty} 
	= \frac{-1}{-i \omega +\lambda_m} \,,	\nonumber 	\\ 
\mathrm{Re} \int_0^{\infty} d\tau e^{(-i \omega +\lambda_m) \tau} 
	&=& - \mathrm{Re} \left[ \frac{i \omega +\lambda_m}
	{ \omega^2 +\lambda_m^2} \right] 
	= \frac{- \lambda_m}{ \omega^2 +\lambda_m^2} \,. \nonumber \\
\end{eqnarray}
%%

For numerical work (and for any $\Delta$) a matrix analysis is more 
convenient, formally integrating the correlation written as
Eq.(\ref{eq:FlucCorrSol}), 
%
\begin{eqnarray} 	%\label{eq:}
S_{inc}(\omega) 
&=& \frac{1}{ \pi \alpha_{ee}  } \mathrm{Re} \left[  \int_0^{\infty} 
	d\tau e^{-(i \omega \mathbf{1} -\mathbf{M}) \tau} 	
	\langle \Delta \sigma_+ \Delta \mathbf{s} \rangle_{st} \right]_1   
	\nonumber \\ 
&=& \frac{1}{ \pi \alpha_{ee}  } \mathrm{Re} \left[ 
	-(i \omega \mathbf{1} -\mathbf{M})^{-1} 
	e^{-(i \omega \mathbf{1} -\mathbf{M}) \tau} \left. \right|_0^{\infty}  
	 \right. 	\nonumber \\
&&	\left. \times 
	\langle \Delta \sigma_+ \Delta \mathbf{s} \rangle_{st} \right]_1 	
	\nonumber \\ 
&=& \frac{1}{ \pi \alpha_{ee}  } \mathrm{Re} \left[ 
	(i \omega \mathbf{1} -\mathbf{M})^{-1} 
	\langle \Delta \sigma_+ \Delta \mathbf{s} \rangle_{st} \right]_1 	\,, 	
	\nonumber \\ 
\end{eqnarray}
%%
where $\mathbf{1}$ is the $4 \times 4$ identity matrix. The subindex $1$ 
outside the square brackets indicates that only the first element of the 
resulting vector is to be taken.

\subsection{Integrated Spectra} 
Now, we sketch the calculation of the intensity of a peak. First, the 
components of the incoherent spectra can be well approximated by 
Lorentzians, 
%
\begin{eqnarray} 
S_m(\omega) = \frac{\lambda_m}{\omega^2 +\lambda_m^2} 
	= \frac{- a_m}{\omega^2 +a_m^2} 	\,
\end{eqnarray}
%%
since the eigenvalues (\ref{eq:eigenvalues}) have negative real parts, 
$\lambda_m = -a_m$. Then, we integrate over all frequencies, 
%
\begin{eqnarray} 
\int_{-\infty}^{\infty} \frac{d\omega}{\omega^2 +a^2} 
	&=& \left. \frac{1}{a} \tan^{-1}{\left( \frac{\omega}{a} \right)} 
		\right|_{-\infty}^{\infty} 
	=  \frac{1}{a} \left[ \frac{\pi}{2} -\frac{-\pi}{2} \right] 	\nonumber \\ 
	&=& \pi/a	\,.
\end{eqnarray}

\section{Replacing the Laser with an Incoherent Field}
%
%\begin{figure}[b]
%\includegraphics{3LA_incoherent.pdf} 
%\caption{\label{fig:3LA} 
%Scheme of the three-level atom-laser 
%interaction and spontaneous decay rates. }
%\end{figure}
%% 
Here, we revisit the model studied in the previous sections by replacing 
the laser driving the $|g \rangle - |e \rangle$ transition with a broadband 
incoherent field. Instead of the atom-laser Hamiltonian, we have now a 
new term in the Liouvillian describing this kind of excitation that could 
somewhat be interpreted as blackbody radiation of average photon 
number $\bar{n}$. This could be produced by scrambling the phase of 
a laser or if the laser operates below threshold. The spontaneous 
emission scheme is the same as in the laser problem. 

The master equation is 
%
\begin{eqnarray}
\dot{\rho}(t) &=&  \kappa (2\sigma_{eg} \rho \sigma_{ge} 
  	-\sigma_{ge} \sigma_{eg} \rho  -\rho \sigma_{ge} \sigma_{eg})     
  	\nonumber \\
&&+\frac{\gamma}{2} (2\sigma_{ge} \rho \sigma_{eg} 
  	-\sigma_{eg} \sigma_{ge} \rho  -\rho \sigma_{eg} \sigma_{ge})     
  	\nonumber \\
&& + \frac{\gamma_d}{2} (2\sigma_{ae} \rho \sigma_{ea} 
  	-\sigma_{ea} \sigma_{ae} \rho -\rho \sigma_{ea} \sigma_{ae}) 	
	\nonumber \\ 
&& +\frac{\gamma}{2} (2\sigma_{ga} \rho \sigma_{ag} 
  	-\sigma_{ag} \sigma_{ga} \rho  -\rho \sigma_{ag} \sigma_{ga}) \,, 
\end{eqnarray}
%%
where $\kappa = \gamma \bar{n}/2$ is the incoherent excitation rate. 
Actually, $\kappa$ would not depend on $\bar{n}$ if the reservoir is at 
zero temperature, so let us make it an independent rate parameter. 
The master equation can be simplified as 
%
\begin{eqnarray}
\dot{\rho}(t) &=& \kappa ( 2 \rho_{gg} \sigma_{ee} -\sigma_{gg} \rho 
	-\rho \sigma_{gg} ) 	\nonumber \\
&& -\frac{\gamma_+}{2} (\sigma_{ee} \rho +\rho \sigma_{ee}) 
    -\frac{\gamma_a}{2} (\sigma_{aa} \rho +\rho \sigma_{aa}) 
        \nonumber \\
&& +\gamma \rho_{ee} \sigma_{gg}  +\gamma_d \rho_{ee} \sigma_{aa}    
    +\gamma_a \rho_{aa} \sigma_{gg}  \,.     \label{masterEqInc}
\end{eqnarray}
%%

The equations for the populations are decoupled from those of the 
coherences. Moreover, the latter are zero at all times. The relevant 
equations are ($\gamma_+ = \gamma + \gamma_d$)
%
\begin{subequations}
\begin{eqnarray}
\dot{\rho}_{ee} &=& 2\kappa \rho_{gg} -\gamma_+ \rho_{ee} 	\,, \\ 
\dot{\rho}_{gg} &=& -2\kappa \rho_{gg} +\gamma \rho_{ee} 
	+\gamma_a \rho_{aa} 	\,, \\  
\dot{\rho}_{aa} &=& \gamma_d \rho_{ee} -\gamma_a \rho_{aa} 	\,.
\end{eqnarray}
\end{subequations}
%%
Using the conservation of population, $\rho_{aa} = 1- \rho_{ee} -\rho_{gg}$, 
the system of equations is reduced to 
%
\begin{subequations}
\begin{eqnarray}
\dot{\rho}_{ee} &=& 2\kappa \rho_{gg} -\gamma_+ \rho_{ee} 	\,, \\ 
\dot{\rho}_{gg} &=& -(2\kappa +\gamma_a) \rho_{gg} 
	+\gamma_- \rho_{ee} +\gamma_a  	 	\,.
\end{eqnarray}
\end{subequations}
%%
where $\gamma_- = \gamma -\gamma_a$. 

These equations can be written in matrix form:
%
\begin{eqnarray}
\dot{\mathbf{r}}(t) = \mathbf{M} \mathbf{r}(0) +\mathbf{b} 	\,,
\end{eqnarray}
%%
where 
%
\begin{eqnarray}
\mathbf{r} = \left( \begin{array}{c} \rho_{ee} \\ \rho_{gg} \end{array}  \right) \,, 
	\qquad \qquad
\mathbf{b} = \left( \begin{array}{c} 0 \\ \gamma_a \end{array}  \right) \,, \\
\mathbf{M} = \left( \begin{array}{cc} -\gamma_+ & 2\kappa \\ 
	\gamma_- & -(2\kappa +\gamma_a)  \end{array}  \right) \,.  
\end{eqnarray}
%%
 
The steady state solutions (where $\dot{\rho}_{jj} =0$) are: 
%
\begin{subequations}
\begin{eqnarray}
\rho_{ee}^{st} &=& \frac{ 2\kappa \gamma_a }
		{ \gamma_+ \gamma_a +2\kappa (\gamma_d +\gamma_a) } \,, \\
\rho_{gg}^{st} &=& \frac{ \gamma_+ \gamma_a }
		{ \gamma_+ \gamma_a +2\kappa (\gamma_d +\gamma_a) } \,, \\ 
\rho_{aa}^{st} &=& \frac{ 2\kappa \gamma_d }
		{ \gamma_+ \gamma_a +2\kappa (\gamma_d +\gamma_a) } \,. 
\end{eqnarray}
\end{subequations}
%%
It can be verified that the populations add to one.

$\mathbf{M}$ are found to be

	&&\pm \frac{1}{2} 
	\sqrt{ 8\kappa \gamma_- +(2\kappa -\gamma_+ +\gamma_a)^2  } \,, \\
\lambda_+ &\ll& \lambda_- \,.  
\end{eqnarray}
\end{subequations}
